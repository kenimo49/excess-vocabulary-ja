def plot_pes(self):
    """PES曲線をプロット"""
    fig, ax = plt.subplots(figsize=(10, 6))
    ax.plot(self.distances, self.energies, 'bo-', linewidth=2, markersize=6)
    min_idx = np.argmin(self.energies)
    ax.scatter(
        [self.distances[min_idx]], [self.energies[min_idx]],
        color='red', s=200, zorder=5,
        label=f'Equilibrium: r={self.distances[min_idx]:.3f} Bohr'
    )
    ax.set_xlabel('Bond Length (Bohr)', fontsize=12)
    ax.set_ylabel('Energy (Hartree)', fontsize=12)
    ax.set_title(f'Potential Energy Surface: {self.base_molecule.name}', fontsize=14)
    ax.legend(); ax.grid(True, alpha=0.3)
    plt.tight_layout()
    plt.savefig('pes_scan.png', dpi=150)
    return fig
