    def build_qaoa_circuit(self, p=3):
        @qml.qnode(self.device)
        def qaoa_circuit(gamma_params, beta_params):
            for i in range(self.n_qubits):
                qml.Hadamard(wires=i)  # |+>^n 初期化
            for layer in range(p):
                self._qaoa_cost_layer(gamma_params[layer])
                self._qaoa_mixer_layer(beta_params[layer])
            return sum(
                d.get('weight',1.0) * (1 - qml.expval(qml.PauliZ(i) @ qml.PauliZ(j))) / 2
                for i, j, d in self.task_graph.edges(data=True)
            )
        return qaoa_circuit

    def optimize_schedule(self, p=3, max_steps=100):
        if self.task_graph is None:
            self.build_conflict_graph()
        qaoa = self.build_qaoa_circuit(p)
        np.random.seed(42)
        gamma = np.random.uniform(0, np.pi, size=p)
        beta  = np.random.uniform(0, np.pi/2, size=p)
        params = np.concatenate([gamma, beta])

        def cost_fn(params):
            return -qaoa(params[:p], params[p:])  # 最大化のため負号

        opt = qml.GradientDescentOptimizer(stepsize=0.1)
        for step in range(max_steps):
            params, cost = opt.step_and_cost(cost_fn, params)
            if step % 20 == 0:
                print(f"Step {step}: Cut={-float(cost):.4f}")

        best_asgn, best_cut = self._sample_best_assignment(qaoa, params[:p], params[p:])
        return {
            "assignment": best_asgn,
            "cut_value": best_cut,
            "schedule": self._interpret_assignment(best_asgn)
        }
