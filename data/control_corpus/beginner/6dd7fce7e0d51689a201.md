def build_cost_function(self):
    @qml.qnode(self.device, diff_method="adjoint")
    def cost_fn(params):
        self._ansatz_circuit(params, list(range(self.n_qubits)))
        return qml.expval(self.hamiltonian)
    return cost_fn

def optimize(self, max_steps=200, lr=0.3):
    """
    Adam最適化でVQEを実行
    Returns: float: 基底状態エネルギー（ハートリー単位）
    """
    if self.hamiltonian is None:
        self.build_hamiltonian()
    np.random.seed(42)
    params = np.random.uniform(-np.pi, np.pi, (self.n_layers, self.n_qubits, 3))
    cost_fn = self.build_cost_function()
    optimizer = qml.AdamOptimizer(stepsize=lr)
    prev_energy = float('inf')
    convergence_threshold = 1e-6
    for step in range(max_steps):
        params, energy = optimizer.step_and_cost(cost_fn, params)
        self.cost_history.append(float(energy))
        if step % 20 == 0:
            print(f"Step {step:4d}: E = {energy:.8f} Ha")
        if abs(energy - prev_energy) < convergence_threshold:
            print(f"収束達成 (Step {step})")
            break
        prev_energy = energy
    print(f"最終エネルギー: {float(energy):.8f} Ha = {float(energy)*27.2114:.4f} eV")
    return float(energy)

def plot_convergence(self, reference_energy=None):
    fig, ax = plt.subplots(figsize=(10, 5))
    ax.plot(range(len(self.cost_history)), self.cost_history, 'b-', linewidth=2, label='VQE Energy')
    if reference_energy:
        ax.axhline(y=reference_energy, color='r', linestyle='--',
                  label=f'FCI Reference: {reference_energy:.4f} Ha')
    ax.set_xlabel('Optimization Step'); ax.set_ylabel('Energy (Hartree)')
    ax.set_title(f'VQE Convergence: {self.molecule.name}')
    ax.legend(); ax.grid(True, alpha=0.3)
    plt.tight_layout(); plt.savefig('vqe_convergence.png', dpi=150)
    return fig
