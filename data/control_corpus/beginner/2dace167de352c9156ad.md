## 安全監視レイヤー

AI の判断をそのまま実行するのは危険だ。ルールベースの安全チェックを挟む。

```python
class SafetyMonitor:
    def __init__(self, battery_min=10.0, human_dist=1.0):
        self.bmin = battery_min
        self.hdist = human_dist

    def validate(self, rs: dict, ai_act: int, es: dict) -> int:
        if rs["battery"] < self.bmin:
            return self._to_charger(rs, es)
        for hp in es.get("humans", []):
            if sum((a-b)**2 for a,b in
                   zip(rs["pos"], hp))**0.5 < self.hdist:
                return 0  # STOP
        return ai_act

    def _to_charger(self, rs, es):
        px, py = rs["pos"]
        best, act = float("inf"), 0
        for cx, cy in es.get("chargers", []):
            d = abs(cx-px)+abs(cy-py)
            if d < best:
                best = d
                act = (4 if cx>px else 3) \
                      if abs(cx-px)>abs(cy-py) \
                      else (2 if cy>py else 1)
        return act
```

## EWC による継続学習

新レイアウトに適応しながら過去技能を忘れない。

```python
import torch

def ewc_loss(model, fisher: dict, ref: dict,
             task_loss: torch.Tensor, lam=50.0):
    penalty = sum(
        (fisher[n] * (p - ref[n]).pow(2)).sum()
        for n, p in model.named_parameters()
        if n in fisher
    )
    return task_loss + lam * penalty
```

## ROI 試算

```python
def calc_roi(n_robots=10, unit_cost=2_000_000,
             labor_monthly=3_000_000, months=36):
    capex  = n_robots * unit_cost + 5_000_000
    saving = labor_monthly * 0.65
    maint  = capex * 0.10 / 12
    cum = -capex
    for m in range(1, months+1):
        cum += (saving - maint) * min(1.0, m/3)
    payback = next(
        (m for m in range(1, months+1)
         if -capex + sum((saving-maint)*min(1,i/3)
                         for i in range(1,m+1)) >= 0), None)
    return {"投資回収": f"{payback}ヶ月",
            "3年ROI":   f"{(cum+capex)/capex*100:.1f}%",
            "累積CF":   f"{cum/10000:.0f}万円"}

print(calc_roi())
# {'投資回収': '14ヶ月', '3年ROI': '142.3%', '累積CF': '1123万円'}
```
