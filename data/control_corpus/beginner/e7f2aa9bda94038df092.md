## 3. VQEによる医薬品分子エネルギー計算

### 3.1 ハミルトニアンの構築

水素分子（H₂）を例に、薬物分子の結合エネルギー計算を実装します。

```python
"""
VQE（変分量子固有値ソルバー）による分子エネルギー計算
医薬品開発での分子設計基盤となる実装
"""

import pennylane as qml
from pennylane import numpy as np
import matplotlib.pyplot as plt
from dataclasses import dataclass
from typing import List, Tuple, Optional


@dataclass
class MoleculeConfig:
    """分子設定クラス"""
    name: str
    symbols: List[str]
    coordinates: np.ndarray
    charge: int = 0
    mult: int = 1
    basis: str = "sto-3g"


class VQEMolecularSolver:
    """
    VQEを用いた分子エネルギー計算クラス

    医薬品開発における分子軌道計算を量子コンピュータで実行する。
    特に薬物候補分子の安定性評価（結合エネルギー）に有用。
    """

    def __init__(self, molecule: MoleculeConfig, n_layers: int = 2):
        self.molecule = molecule
        self.n_layers = n_layers
        self.hamiltonian = None
        self.n_qubits = None
        self.device = None
        self.cost_history = []

    def build_hamiltonian(self) -> Tuple:
        """
        分子ハミルトニアンをパウリ演算子形式で構築

        Returns:
            Tuple[Hamiltonian, int]: ハミルトニアンと必要量子ビット数
        """
        molecule = qml.qchem.Molecule(
            symbols=self.molecule.symbols,
            coordinates=self.molecule.coordinates,
            charge=self.molecule.charge,
            mult=self.molecule.mult,
            basis_name=self.molecule.basis
        )

        # フェルミオン-量子ビット変換（Jordan-Wigner変換）
        hamiltonian, n_qubits = qml.qchem.molecular_hamiltonian(
            molecule,
            method="dhf",  # Differentiable Hartree-Fock
            active_electrons=2,
            active_orbitals=2
        )

        self.hamiltonian = hamiltonian
        self.n_qubits = n_qubits
        self.device = qml.device("default.qubit", wires=n_qubits)

        print(f"分子: {self.molecule.name}")
        print(f"必要量子ビット数: {n_qubits}")
        print(f"ハミルトニアン項数: {len(hamiltonian.ops)}")

        return hamiltonian, n_qubits

    def _ansatz_circuit(self, params: np.ndarray, wires: List[int]):
        n_wires = len(wires)
        qml.BasisState(np.array([1, 1, 0, 0]), wires=wires)
        for layer in range(self.n_layers):
            for i, wire in enumerate(wires):
                qml.RZ(params[layer, i, 0], wires=wire)
                qml.RY(params[layer, i, 1], wires=wire)
                qml.RZ(params[layer, i, 2], wires=wire)
            for i in range(n_wires - 1):
                qml.CNOT(wires=[wires[i], wires[i + 1]])
            if n_wires > 2:
                qml.CNOT(wires=[wires[-1], wires[0]])
```
