import matplotlib.pyplot as plt
from matplotlib.patches import Patch

def visualize_schedule(scheduler, assignment):
    G = scheduler.task_graph
    pos = nx.spring_layout(G, seed=42)
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(14, 6))

    # 元の競合グラフ
    nx.draw(G, pos, ax=ax1, with_labels=True,
            node_color='lightblue', node_size=800,
            labels={i: f"T{i+1}" for i in G.nodes()})
    ax1.set_title("タスク競合グラフ")

    # 最適割り当て後
    colors = ['#FF6B6B' if a==0 else '#4ECDC4' for a in assignment]
    nx.draw(G, pos, ax=ax2, node_color=colors, node_size=800, with_labels=True,
            labels={i: f"T{i+1}" for i in G.nodes()})
    # カットエッジを強調
    cut_edges = [(i,j) for i,j in G.edges() if assignment[i]!=assignment[j]]
    nx.draw_networkx_edges(G, pos, edgelist=cut_edges, ax=ax2,
                           edge_color='gray', style='dashed', width=2)
    ax2.set_title("QAOA最適化結果\n(赤=ロボット1, 青緑=ロボット2)")
    ax2.legend(handles=[
        Patch(facecolor='#FF6B6B', label='Robot 1'),
        Patch(facecolor='#4ECDC4', label='Robot 2')
    ])
    plt.tight_layout()
    plt.savefig('robot_schedule.png', dpi=150)
    return fig
