# 【物流AI #2】PPO で倉庫ロボットを学習させる：Actor-Critic 実装

## Actor-Critic ネットワーク

倉庫ロボットは「カメラ画像」と「関節角度・力覚」の2種類の入力を持つ。これらを融合する構造が重要だ。

```python
import torch, torch.nn as nn

class RobotAC(nn.Module):
    def __init__(self, proprio_dim: int, act_dim: int):
        super().__init__()
        # カメラ画像エンコーダ
        self.vis = nn.Sequential(
            nn.Conv2d(3, 32, 8, 4), nn.ReLU(),
            nn.Conv2d(32, 64, 4, 2), nn.ReLU(),
            nn.AdaptiveAvgPool2d(4), nn.Flatten(),
            nn.Linear(1024, 256), nn.LayerNorm(256)
        )
        # 固有感覚エンコーダ
        self.prop = nn.Sequential(
            nn.Linear(proprio_dim, 128), nn.ReLU(),
            nn.LayerNorm(128)
        )
        # 融合 → 出力
        self.trunk  = nn.Sequential(nn.Linear(384, 256), nn.ReLU())
        self.actor  = nn.Linear(256, act_dim)
        self.critic = nn.Linear(256, 1)
        self.log_std = nn.Parameter(torch.full((act_dim,), -0.5))

    def forward(self, img, prop):
        h = self.trunk(torch.cat([self.vis(img),
                                   self.prop(prop)], -1))
        return self.actor(h), self.critic(h), self.log_std.exp()
```

## PPO 更新（クリッピング損失）

```python
def ppo_step(model, opt, batch, clip=0.2):
    img, prop, act, old_lp, adv, ret = batch
    mu, val, std = model(img, prop)
    dist   = torch.distributions.Normal(mu, std)
    new_lp = dist.log_prob(act).sum(-1)
    ent    = dist.entropy().sum(-1).mean()

    r   = (new_lp - old_lp).exp()
    adv = (adv - adv.mean()) / (adv.std() + 1e-8)
    L_pi = -torch.min(r * adv,
                      r.clamp(1-clip, 1+clip) * adv).mean()
    L_v  = (val.squeeze() - ret).pow(2).mean()
    loss = L_pi + 0.5 * L_v - 0.01 * ent

    opt.zero_grad(); loss.backward()
    nn.utils.clip_grad_norm_(model.parameters(), 0.5)
    opt.step()
    return {"pi": L_pi.item(), "v": L_v.item(), "ent": ent.item()}
```

## GAE でアドバンテージを計算

```python
def compute_gae(rews, vals, dones, last_val,
                gamma=0.99, lam=0.95):
    adv, g = [], 0.0
    for t in reversed(range(len(rews))):
        nv  = last_val if t == len(rews)-1 else vals[t+1]
        d   = rews[t] + gamma * nv * (1-dones[t]) - vals[t]
        g   = d + gamma * lam * (1-dones[t]) * g
        adv.insert(0, g)
    adv = torch.tensor(adv)
    return adv, adv + torch.tensor(vals)
```

## 学習安定化のチェックリスト

- `clip_grad_norm_` でグラジエントを 0.5 以下にクリップ
- アドバンテージを正規化（平均0・分散1）
- 学習率は `3e-4`、エポック数は `4〜10`
- エントロピー係数 `0.01` で探索を維持
