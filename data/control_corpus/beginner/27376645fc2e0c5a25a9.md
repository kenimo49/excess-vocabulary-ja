def run_h2_vqe():

        name="H2",

    print(f"FCI参照値との誤差: {error:.2f} mHa")
    solver.plot_convergence(reference_energy=fci_reference)
    return energy

### 3.2 ポテンシャルエネルギー曲面スキャン（PES Scan）

```python
class PESScanner:
    """分子間距離を変えながらVQEエネルギーをスキャン"""
    def __init__(self, base_molecule):
        self.base_molecule = base_molecule
        self.distances = []; self.energies = []

    def scan_bond_length(self, r_min=0.5, r_max=3.0, n_points=20, n_layers=2):
        distances = np.linspace(r_min, r_max, n_points)
        for i, r in enumerate(distances):
            new_coords = self.base_molecule.coordinates.copy()
            new_coords[1] = np.array([0.0, 0.0, r])
            mol_config = MoleculeConfig(
                name=f"{self.base_molecule.name}(r={r:.2f})",
                symbols=self.base_molecule.symbols,
                coordinates=new_coords,
            )
            solver = VQEMolecularSolver(mol_config, n_layers=n_layers)
            solver.build_hamiltonian()
            e = solver.optimize(max_steps=100, lr=0.3)
            self.energies.append(e)
            print(f"[{i+1}/{n_points}] r={r:.3f}: E={e:.6f} Ha")
        self.distances = distances.tolist()
        return self.distances, self.energies

    def find_equilibrium(self):
        min_i = np.argmin(self.energies)
        d, e = self.distances[min_i], self.energies[min_i]
        print(f"平衡結合距離: {d:.4f} Bohr ({d*0.529177:.4f} A)")
        print(f"最小エネルギー: {e:.8f} Ha")
        return d, e
```
