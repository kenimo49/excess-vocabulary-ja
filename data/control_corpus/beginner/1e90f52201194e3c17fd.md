    def _sample_best_assignment(self, circuit, gamma, beta, n_samples=1000):
        @qml.qnode(self.device)
        def sampling_circuit(gamma, beta):
            for i in range(self.n_qubits): qml.Hadamard(wires=i)
            for l in range(len(gamma)):
                self._qaoa_cost_layer(gamma[l])
                self._qaoa_mixer_layer(beta[l])
            return qml.probs(wires=range(self.n_qubits))

        probs = sampling_circuit(gamma, beta)
        best_cut, best_asgn = -1, None
        for idx in np.argsort(probs)[-20:]:  # 上位20候補を評価
            asgn = [int(b) for b in format(idx, f'0{self.n_qubits}b')]
            cut = sum(
                d.get('weight',1.0)
                for i,j,d in self.task_graph.edges(data=True) if asgn[i]!=asgn[j]
            )
            if cut > best_cut:
                best_cut, best_asgn = cut, asgn
        return best_asgn, best_cut

    def _interpret_assignment(self, assignment):
        r1 = [i for i,a in enumerate(assignment) if a==0]
        r2 = [i for i,a in enumerate(assignment) if a==1]
        print(f"ロボット1: タスク {[t+1 for t in r1]}")
        print(f"ロボット2: タスク {[t+1 for t in r2]}")
        return {"robot_1": r1, "robot_2": r2}
