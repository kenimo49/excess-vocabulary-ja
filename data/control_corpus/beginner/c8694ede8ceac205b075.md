class HybridQNN(nn.Module):
    def __init__(self, n_features, n_classes=2, n_qubits=N_QUBITS, n_layers=N_LAYERS, dropout=0.2):
        super().__init__()
        self.classical_pre = nn.Sequential(
            nn.Linear(n_features, 64), nn.BatchNorm1d(64), nn.ReLU(),
            nn.Dropout(dropout), nn.Linear(64, 32), nn.ReLU(),
            nn.Linear(32, n_qubits), nn.Tanh()
        )
        self.quantum_layer = QuantumLayer(n_qubits=n_qubits, n_layers=n_layers)
        self.classical_post = nn.Sequential(
            nn.Linear(n_qubits, 16), nn.ReLU(),
            nn.Dropout(dropout), nn.Linear(16, n_classes)
        )
        self.softmax = nn.Softmax(dim=1)

    def forward(self, x):
        x = self.classical_pre(x)           # 古典圧縮
        x = self.quantum_layer(x)           # 量子計算
        return self.softmax(self.classical_post(x.float()))

    def get_quantum_state_info(self, x):
        """量子期待値とエンタングルメント指標を返す（解釈性用）"""
        with torch.no_grad():
            xc = self.classical_pre(x)
            qo = self.quantum_layer(xc)
        corr = np.corrcoef(qo.numpy().T)
        off = corr[np.triu_indices_from(corr, k=1)]
        return {
            "quantum_expectation_values": qo.numpy(),
            "entanglement_entropy": float(np.mean(np.abs(off)))
        }

**量子状態情報の活用例：**
- `entanglement_entropy` が高い → 量子回路がデータの複雑な相関を捕捉
- 各量子ビットの期待値の分布 → 特徴重要度の代替指標
- 推論可視化ダッシュボードへの組み込みが可能
