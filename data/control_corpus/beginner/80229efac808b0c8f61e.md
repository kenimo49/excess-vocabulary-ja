## 6. 量子転移学習（Quantum Transfer Learning）

### 6.1 古典モデルと量子層の融合

事前学習済みResNet18の特徴抽出層を活かしつつ、分類ヘッドを量子回路に置き換えます。

```python
import torch, torch.nn as nn, pennylane as qml, numpy as np
from torchvision import models

class QuantumTransferLearning(nn.Module):
    """
    量子転移学習モデル
    ResNet18 → 特徴抽出 → 量子回路 → 分類
    応用: 顕微鏡画像からの細胞薬物応答分類
    """
    def __init__(self, n_qubits=4, n_qlayers=6, n_classes=4, freeze_classical=True):
        super().__init__()
        # 古典バックボーン
        self.backbone = models.resnet18(weights='IMAGENET1K_V1')
        if freeze_classical:
            for p in self.backbone.parameters(): p.requires_grad = False
        feat_dim = self.backbone.fc.in_features  # 512
        self.backbone.fc = nn.Linear(feat_dim, n_qubits)
        self.backbone.fc.requires_grad_(True)
        # 量子層
        dev_q = qml.device("default.qubit", wires=n_qubits)
        @qml.qnode(dev_q, interface="torch", diff_method="adjoint")
        def q_circuit(inputs, weights):
            qml.AngleEmbedding(inputs * np.pi, wires=range(n_qubits), rotation='Y')
            qml.StronglyEntanglingLayers(weights, wires=range(n_qubits))
            return [qml.expval(qml.PauliZ(i)) for i in range(n_qubits)]
        weight_shapes = {"weights": (n_qlayers, n_qubits, 3)}
        self.quantum_layer = qml.qnn.TorchLayer(q_circuit, weight_shapes)
        # 分類ヘッド
        self.classifier = nn.Sequential(
            nn.Linear(n_qubits, 16), nn.ReLU(), nn.Dropout(0.3),
            nn.Linear(16, n_classes)
        )
    def forward(self, x):
        features = torch.tanh(self.backbone(x))  # [-1,1]正規化
        q_features = self.quantum_layer(features)
        return self.classifier(q_features.float())
```
