    # 訓練ループ（ManufacturingQMLPipeline.train の続き）
    for epoch in range(epochs):
        self.model.train()
        epoch_loss = 0.0
        for i in range(n_batches):
            s, e = i*batch_size, (i+1)*batch_size
            optimizer.zero_grad()
            out = self.model(X_tr_t[s:e])
            loss = criterion(out, y_tr_t[s:e])
            loss.backward(); optimizer.step()
            epoch_loss += loss.item()
        scheduler.step()
        if (epoch + 1) % 5 == 0:
            self.model.eval()
            with torch.no_grad():
                v_out = self.model(X_v_t)
                v_loss = criterion(v_out, y_v_t).item()
                v_acc = (v_out.argmax(1)==y_v_t).float().mean().item()
            print(f"Epoch {epoch+1:3d}: Loss={epoch_loss/n_batches:.4f} ValLoss={v_loss:.4f} ValAcc={v_acc:.4f}")

    def evaluate(self, X_test, y_test):
        X_t = self.scaler.transform(X_test)
        X_t = torch.FloatTensor(X_t)
        self.model.eval()
        with torch.no_grad():
            out = self.model(X_t)
            preds = out.argmax(1).numpy()
            probs = out[:,1].numpy()
        acc = (preds==y_test).mean()
        auc = roc_auc_score(y_test, probs)
        print(f"Accuracy: {acc:.4f} | ROC-AUC: {auc:.4f}")
        print(classification_report(y_test, preds, target_names=['正常品','不良品']))
        return {"accuracy": acc, "roc_auc": auc}
