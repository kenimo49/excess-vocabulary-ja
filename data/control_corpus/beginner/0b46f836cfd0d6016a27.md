Option Explicit

Sub AggregateIP()

    Dim wsSrc As Worksheet, wsDst As Worksheet
    Dim lastRow As Long, i As Long
    Dim dict As Object
    Dim key As Variant
    Dim ip As String, name1 As String, name2 As String
    Dim arr As Variant
    
    Set wsSrc = ThisWorkbook.Sheets("Sheet1")
    Set wsDst = ThisWorkbook.Sheets("Sheet2")
    Set dict = CreateObject("Scripting.Dictionary")
    
    lastRow = wsSrc.Cells(wsSrc.Rows.Count, "A").End(xlUp).Row
    
    For i = 2 To lastRow
        ip = wsSrc.Cells(i, 1).Value
        name1 = wsSrc.Cells(i, 2).Value
        name2 = wsSrc.Cells(i, 3).Value
        
        If Not dict.Exists(ip) Then
            dict.Add ip, Array(1, name1, name2)
        Else
            arr = dict(ip)
            arr(0) = arr(0) + 1
            dict(ip) = arr
        End If
    Next i
    
    wsDst.Cells.ClearContents
    wsDst.Range("A1:D1").Value = Array("IP", "count", "name1", "name2")
    
    i = 2
    For Each key In dict.Keys
        wsDst.Cells(i, 1).Value = key
        wsDst.Cells(i, 2).Value = dict(key)(0)
        wsDst.Cells(i, 3).Value = dict(key)(1)
        wsDst.Cells(i, 4).Value = dict(key)(2)
        i = i + 1
    Next key

    MsgBox "集計が完了しました。", vbInformation

End Sub
