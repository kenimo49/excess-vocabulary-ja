# 【物流AI #5】マルチエージェントRLで複数AGVを協調させる最小実装

## パラメータ共有 MARL の考え方

倉庫に AGV が 10 台いる場合、各 AGV に別々のネットワークを持たせると学習に膨大なデータが必要になる。**全 AGV が同一ネットワークを共有**（パラメータ共有）すれば、実質 10 倍のサンプルで学習できる。

## グリッドワールド環境（最小実装）

```python
import numpy as np
from collections import defaultdict

class MiniWarehouse:
    MOVES = {0:(0,0),1:(0,-1),2:(0,1),3:(-1,0),4:(1,0)}

    def __init__(self, W=15, H=15, n_bots=3):
        self.W, self.H, self.nb = W, H, n_bots
        self.reset()

    def reset(self):
        self.bots  = {i: np.random.randint([0,0],[self.W,self.H]) for i in range(self.nb)}
        self.goals = {i: np.random.randint([0,0],[self.W,self.H]) for i in range(self.nb)}
        self.t = 0
        return {i: np.concatenate([p/[self.W,self.H],
                (self.goals[i]-p)/[self.W,self.H]])
                for i, p in self.bots.items()}

    def step(self, actions):
        rews, new_pos = {i:0.0 for i in range(self.nb)}, {}
        for i, a in actions.items():
            dx, dy = self.MOVES[a]
            new_pos[i] = np.clip(self.bots[i]+[dx,dy],[0,0],[self.W-1,self.H-1])
        occ = defaultdict(list)
        for i,p in new_pos.items(): occ[tuple(p)].append(i)
        for p, ids in occ.items():
            if len(ids)>1:
                for i in ids: new_pos[i]=self.bots[i].copy(); rews[i]-=1.0
        for i in range(self.nb):
            self.bots[i] = new_pos[i]
            d = np.linalg.norm(self.bots[i]-self.goals[i])
            rews[i] -= 0.01*d
            if d < 1.0:
                rews[i] += 10.0
                self.goals[i] = np.random.randint([0,0],[self.W,self.H])
        self.t += 1
        obs = {i: np.concatenate([p/[self.W,self.H],
               (self.goals[i]-p)/[self.W,self.H]])
               for i, p in self.bots.items()}
        return obs, rews, self.t>=300, {}
```

## 共有エージェントと訓練ループ

```python
import torch, torch.nn as nn

class SharedAgent(nn.Module):
    def __init__(self, obs=4, acts=5, h=64):
        super().__init__()
        self.net = nn.Sequential(nn.Linear(obs,h),nn.ReLU(),
                                  nn.Linear(h,h),nn.ReLU())
        self.pi = nn.Sequential(nn.Linear(h,acts),nn.Softmax(-1))

    def act(self, o, eps=0.1):
        if np.random.random() < eps: return np.random.randint(5)
        with torch.no_grad():
            return self.pi(self.net(
                torch.FloatTensor(o).unsqueeze(0))).argmax().item()

env, agent = MiniWarehouse(n_bots=3), SharedAgent()
for ep in range(200):
    obs, total = env.reset(), 0.0
    while True:
        acts = {i: agent.act(o) for i,o in obs.items()}
        obs, rews, done, _ = env.step(acts)
        total += sum(rews.values())
        if done: break
    if ep % 50 == 0: print(f"ep {ep:3d}: {total:.1f}")
```

## 設計のポイント

- **パラメータ共有**: 全 AGV が同一重み → サンプル効率 N 倍
- **衝突回避**: ルールベース + ペナルティ → 安全性と学習の両立
- **目標再配置**: 到達後にランダム更新 → 汎化性向上
